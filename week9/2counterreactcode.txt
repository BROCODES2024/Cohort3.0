import { useState } from "react";
function App() {
  return (
    <div>
      <Counter></Counter>
    </div>
  );
}
function Counter() {
  const [count, setCount] = useState(0);
  function inccount() {
    setCount(count + 1);
  }
  function deccount() {
    setCount(count - 1);
  }
  function resetcount() {
    setCount(0);
  }
  return (
    <div>
      <h1 id="text">{count}</h1>
      <button onClick={inccount}>Increase count</button>
      <button onClick={deccount}>Decrease count</button>
      <button onClick={resetcount}>Reset count</button>
    </div>
  );
}
export default App;
